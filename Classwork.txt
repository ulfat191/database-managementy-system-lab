select First_Name, Department_Name, 
Employees.Employee_Id, Departments.Department_id from dbo.Employees
join dbo.Departments
on (dbo.Employees.Employee_Id = dbo.Departments.Department_id)

--renaming of instace needed when we join same tables with each other

select E.First_name as 'Employe_Name', M.First_name as
'Manager Name'
from dbo.Employees E, dbo.Employees M
where E.Employee_Id = M.Manager_id
--same table joining
--or equal join
select concat(E.First_name,  ' is manager of ', M.First_name)
'Manager Name'
from dbo.Employees E, dbo.Employees M
where E.Employee_Id = M.Manager_id

--outer join: Accept Null value, if there is o relation return it 
--with NULL value
--Left order join, if there is no match then left table columns
--will stay and right column will be nULL
--Right order join: if there is no match then, right table column
--will stay but left one will be NULL

create table Emp
(
   E_ID nvarchar(100), 
   D_ID nvarchar(100), 
)

create table DEPT
(
    E_ID nvarchar(100), 
	I_ID nvarchar(100)
)

create table LOC
(
  L_ID nvarchar(100),
  L_NAME nvarchar(100)
)

insert into dbo.DEPT values
('A','X'), ('D','Y'), ('E','C')
insert into dbo.LOC values
('X','Dhaka'), ('B','Rajshahi'), ('S','Chattogram')

select * from dbo.Emp
Left join dbo.DEPT
on(dbo.Emp.D_ID=dbo.DEPT.I_ID)

select * from dbo.Emp
Right join dbo.DEPT
on(dbo.Emp.D_ID=dbo.DEPT.I_ID)

select * from dbo.Emp
Left join dbo.DEPT
on(dbo.Emp.D_ID=dbo.DEPT.I_ID)
Right join dbo.LOC
on(dbo.LOC.L_ID=dbo.DEPT.I_ID)

--full order join: Union
select * from dbo.Emp
Left join dbo.DEPT
on(dbo.Emp.D_ID=dbo.DEPT.I_ID)

union
select * from dbo.Emp
Right join dbo.LOC
on(dbo.LOC.L_ID=dbo.DEPT.E_ID)

--union can be done by full order join
